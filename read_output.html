err_tqg = ctqg[:,1] - sigma1

fig, (ax) = plt.subplots(1,3,figsize=(17,4))

ax[0].plot(cqg[:,0], cqg[:,1],'C1--',alpha=0.65)

    Patch(facecolor='C1', edgecolor='k',alpha=0.65, label='QG')]
